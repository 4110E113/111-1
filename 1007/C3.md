## get & fget

```c
#include <stdio.h>

int main(void) {
	char str[20];
	
	puts("請輸入字串 : ");
	gets(str);
	
	puts("請輸入的字串為 : ");
	puts(str);
	
	return 0;
}
```

```c
#include <stdio.h>

int main(void) {
	char str[20];
	
	puts("請輸入字串 : ");
	fgets(str, sizeof(str), stdin);
	
	puts("請輸入的字串為 : ");
	puts(str);
	
	return 0;
}
```

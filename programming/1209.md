#include <stdio.h>
#define ROW 5
#define COLUMN 10


int main(void) {
	int maze[ROW][COLUMN];
	int i, j=0;
	for(i=0; i<ROW; i++){
		for(j=0; j<COLUMN; j++){
			maze[i][j] = (i+1) *(j+1);
	    }
    }
	for(i=0; i<ROW; i++){
		for(j=0; j<COLUMNl j++){
			printf("%d\t", maze[i][j]);
		}
		putchar('\n');
	}
	return 0;
}
